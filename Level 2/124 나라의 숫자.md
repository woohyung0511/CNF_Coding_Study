````java class Solution {
    public String solution(int n) {
        String answer = "";

        while (n != 0) {

            int nmg = n % 3;
            n /= 3;

            if (nmg == 0) {
                nmg = 4;
                n -= 1;
            }
            answer = nmg + answer;
        }
        return answer;
    }
}
````

import java.util.*;

class Solution {
    public String[] solution(String[][] plans) {
        
        //시간기준 오름차순 정렬
        Arrays.sort(plans, (o1, o2) -> timeCul(o1[1]) - timeCul(o2[1]));
        
        //멈춘과제 저장 스택
        Stack<String[]> st = new Stack<>();
        
        //정답 저장 리스트
        ArrayList<String> list = new ArrayList<>();
        
        for (int i = 0; i < plans.length; i++) {
            if (i + 1 < plans.length) {
                int restTime = timeCul(plans[i+1][1]) - (timeCul(plans[i][1]) + Integer.parseInt(plans[i][2]));
                
                //못 끝낸 과제 스택에 추가
                if (restTime < 0) {
                    st.push(new String[]{plans[i][0], plans[i][1], Integer.toString(-restTime)});
                } else {
                    //끝내면 리스트에 추가
                    list.add(plans[i][0]);
                    
                    //시간이 남으면 못 끝낸 과제 하기
                    while (!st.isEmpty() && restTime > 0) {
                        String[] plan = st.pop();
                        int playTime = Integer.parseInt(plan[2]);
                        
                        //끝낼 수 있으면 리스트에 저장
                        if (restTime >= playTime) {
                            list.add(plan[0]);
                            restTime -= playTime;
                        } else {
                            //못 끝내면 남은시간 업데이트
                            st.push(new String[]{plan[0], plan[1], Integer.toString(playTime - restTime)});
                            restTime = 0;
                        }
                    }
                }
            } else if(i+1 == plans.length){
                //마지막 계획 추가
                list.add(plans[i][0]);
            }
        }
        
        //스택에 남아있는 계획들 추가
        while (!st.isEmpty()) {
            list.add(st.pop()[0]);
        }

        //크기가 부족하면 맞는크기 배열 생성
        return list.toArray(new String[0]);
    }

    //시간 -> 분으로 계산
    private int timeCul(String time) {
        String[] hm = time.split(":");
        
        return Integer.parseInt(hm[0]) * 60 + Integer.parseInt(hm[1]);
    }
}
